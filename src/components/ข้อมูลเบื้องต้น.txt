ฐานข้อมูลตอนนี้
Firebase Database Structure:

-Assignments
  - classId: string
  - createdAt: timestamp
  - description: string
  - dueDateTime: string 
  - fileUrl: string
  - id: string
  - points: string
  - title: string

-Attendances
  - activityId: string
  - activityType: string
  - classId: string
  - date: timestamp
  - duration: number
  - studentId: string



- Classrooms
  - ClassId: string
  - ClassName: string
  - ClassDescription: string
  - TeachersID: string (reference to /Teachers/{uid})

- Teachers
  - uid: string
  - FirstName: string
  - LastName: string
  - Username: string
  - Email: string
  - URLImage: string (path to profile image in Firebase Storage)

- Students
  - uid: string
  - FirstName: string
  - LastName: string
  - Username: string
  - Email: string
  - URLImage: string (path to profile image in Firebase Storage)

- ClassEnrollments
  - classId: string (reference to Classrooms)
  - studentId: string (reference to Students)
  - enrollmentDate: timestamp

- Posts
  - id: string
  - classId: string (reference to Classrooms)
  - content: string
  - createdAt: timestamp
  - createdBy: string (reference to Teachers or Students)
  - fileUrl: string (optional)
  - fileName: string (optional)

- Comments
  - id: string
  - postId: string (reference to Posts)
  - classId: string (reference to Classrooms)
  - content: string
  - createdAt: timestamp
  - createdBy: string (reference to Students or Teachers)

- Lessons
  - id: string
  - classId: string (reference to Classrooms)
  - title: string
  - description: string
  - content: string
  - order: number
  - fileUrl: string (optional)
  - fileType: string (optional)

- Quizzes
  - id: string
  - classId: string (reference to Classrooms)
  - title: string
  - description: string
  - questions: array of objects
    - text: string
    - image: string (optional)
    - options: array of objects
      - text: string
      - image: string (optional)
    - correctAnswer: number

- QuizResults
  - id: string (format: {quizId}_{userId})
  - quizId: string (reference to Quizzes)
  - userId: string (reference to Students)
  - score: number
  - totalQuestions: number
  - completedAt: timestamp

- Submissions
  - assignmentId: string
  - classId: string
  - feedback: string
  - fileName: string
  - fileUrl: string
  - grade: number
  - studentId: string
  - submittedAt: timestamp
