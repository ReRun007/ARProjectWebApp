ฐานข้อมูลตอนนี้
// Firebase Database Structure

// Assignments
assignments: {
  [assignmentId]: {
    classId: string,
    createdAt: timestamp,
    description: string,
    dueDateTime: string,
    fileUrl: string,
    points: string,
    title: string
  }
}

// Attendances
attendances: {
  [attendanceId]: {
    activityId: string,
    activityType: string,
    classId: string,
    date: timestamp,
    duration: number,
    studentId: string
  }
}

// Classrooms
classrooms: {
  [classId]: {
    ClassId: string,
    ClassName: string,
    ClassDescription: string,
    TeachersID: string // reference to /teachers/{uid}
  }
}

// Teachers
{
  [uid]: {
    BirthDate: string,
    Email: string,
    FirstName: string,
    Gender: string,
    Institution: string,
    LastName: string,
    URLImage: string, // URL to profile image in Firebase Storage
    UserType: string,
    Username: string
  }
}

// Students
{
  [uid]: {
    BirthDate: string,
    Email: string,
    FirstName: string,
    Gender: string,
    Institution: string,
    LastName: string,
    URLImage: string, // path to profile image in Firebase Storage
    UserType: string,
    Username: string
  }
}

// ClassEnrollments
classEnrollments: {
  [enrollmentId]: {
    classId: string, // reference to classrooms
    studentId: string, // reference to students
    enrollmentDate: timestamp
  }
}

// Posts
posts: {
  [postId]: {
    classId: string, // reference to classrooms
    content: string,
    createdAt: timestamp,
    createdBy: string, // reference to teachers or students
    fileUrl: string, // optional
    fileName: string // optional
  }
}

// Comments
comments: {
  [commentId]: {
    postId: string, // reference to posts
    classId: string, // reference to classrooms
    content: string,
    createdAt: timestamp,
    createdBy: string // reference to students or teachers
  }
}

// Lessons
lessons: {
  [lessonId]: {
    classId: string, // reference to classrooms
    title: string,
    description: string,
    content: string,
    order: number,
    fileUrl: string, // optional
    fileType: string // optional
  }
}

// Quizzes
quizzes: {
  [quizId]: {
    id: string,
    order: number, //ลำดับ
    classId: string, // reference to classrooms
    createdAt: timestamp,
    title: string,
    description: string,
    timeLimit: number, // in minutes
    questions: [
      {
        text: string,
        image: string, // optional
        options: [
          {
            text: string,
            image: string // optional
          }
        ],
        correctAnswer: number
      }
    ]
  }
}

// QuizResults
quizResults: {
  [quizId_userId]: {
    quizId: string, // reference to quizzes
    userId: string, // reference to students
    score: number,
    totalQuestions: number,
    submittedAt: timestamp,
    answers: {
      [questionIndex]: number // selected option index
    }
  }
}

// Submissions
submissions: {
  [submissionId]: {
    assignmentId: string,
    classId: string,
    feedback: string,
    fileName: string,
    fileUrl: string,
    grade: number,
    studentId: string,
    submittedAt: timestamp
  }
}
